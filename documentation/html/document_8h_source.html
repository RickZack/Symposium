50  struct sessionData{
51  friend class boost::serialization::access;
52  unsigned int row, col;
53  std::chrono::system_clock::time_point tmstmp;
54  privilege p;
55  sessionData(privilege p=privilege::none, unsigned int row=0, unsigned int col=0): row{row}, col{col}, p{p}{
56  tmstmp=std::chrono::system_clock::now();
57  }
58  template<class Archive>
59  void serialize(Archive &ar, const unsigned int){
60  ar & row & col & boost::serialization::make_binary_object(&tmstmp, sizeof(tmstmp)) & p;
61  }
62 
63  bool operator==(const sessionData &rhs) const {
64  return row == rhs.row &&
65  col == rhs.col &&
66  tmstmp == rhs.tmstmp &&
67  p == rhs.p;
68  }
69 
70  bool operator!=(const sessionData &rhs) const {
71  return !(rhs == *this);
72  }
